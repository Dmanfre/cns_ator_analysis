season_ator <- ator_only %>% 
  group_by(Season, TeamFull, Conference) %>% 
  summarize(season_team_ator = sum(TeamAssists)/sum(TeamTurnovers),
            season_opp_ator = sum(OpponentAssists)/sum(OpponentTurnovers),
            total_wins = sum(winloss)
            )
